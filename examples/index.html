Entity Framework Core 查询

C# 示例

复杂的数据库查询示例

1public class DeviceService
2{
3    private readonly BMSDataContext _context;
4    
5    public DeviceService(BMSDataContext context)
6    {
7        _context = context;
8    }
9    
10    // 基础查询
11    public async Task<Device?> GetDeviceAsync(int id)
12    {
13        return await _context.Devices.FindAsync(id);
14    }
15    
16    // 复杂查询与投影
17    public async Task<List<DeviceStatsDto>> GetDeviceStatsAsync()
18    {
19        return await _context.Devices
20            .Where(d => d.IsActive)
21            .Select(d => new DeviceStatsDto
22            {
23                Id = d.Id,
24                Name = d.Name,
25                TotalReadings = d.AcurevReadings.Count(),
26                AverageVoltage = d.AcurevReadings.Average(r => r.Voltage),
27                LastReading = d.AcurevReadings
28                    .OrderByDescending(r => r.Timestamp)
29                    .Select(r => r.Timestamp)
30                    .FirstOrDefault()
31            })
32            .ToListAsync();
33    }
34    
35    // 关联查询
36    public async Task<List<Device>> GetDevicesWithBuildingAsync(string buildingName)
37    {
38        return await _context.Devices
39            .Include(d => d.Building)
40            .Include(d => d.AcurevReadings.Take(10))
41            .Where(d => d.Building.Name == buildingName)
42            .ToListAsync();
43    }
44    
45    // 分页查询
46    public async Task<PagedResult<Device>> GetDevicesPagedAsync(int page, int pageSize)
47    {
48        var totalCount = await _context.Devices.CountAsync();
49        
50        var devices = await _context.Devices
51            .OrderBy(d => d.Name)
52            .Skip((page - 1) * pageSize)
53            .Take(pageSize)
54            .ToListAsync();
55            
56        return new PagedResult<Device>
57        {
58            Items = devices,
59            TotalCount = totalCount,
60            PageNumber = page,
61            PageSize = pageSize,
62            TotalPages = (int)Math.Ceiling(totalCount / (double)pageSize)
63        };
64    }
65    
66    // 批量操作
67    public async Task UpdateDeviceStatusAsync(List<int> deviceIds, bool isActive)
68    {
69        await _context.Devices
70            .Where(d => deviceIds.Contains(d.Id))
71            .ExecuteUpdateAsync(d => d.SetProperty(p => p.IsActive, isActive));
72    }
73}
